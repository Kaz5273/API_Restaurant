
```bash
# Clone this repository
$ git clone https://github.com/Kaz5273/API_Restaurant.git

# Go into the back repository and run the app
$ cd API_Restaurant/back-server
$ npm install
$ npm run dev

# Go into the front admin repository and run the app
$ cd API_Restaurant/front-admin
$ npm install
$ npm run dev

# Script that can be run on the back
$ cd API_Restaurant/back-server
$ npm run admin:create                    # Creating an admin
$ npm run user:create                     # Creating a user to order
$ npm run order:create                    # Creating a order
```
